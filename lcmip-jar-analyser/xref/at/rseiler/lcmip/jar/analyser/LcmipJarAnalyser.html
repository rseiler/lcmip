7   import java.io.DataInputStream;

10  import java.util.zip.ZipFile;
11  
12  import static org.objectweb.asm.Opcodes.ASM5;
13  
14  public class LcmipJarAnalyser {
15  
16      public static void main(String[] args) throws IOException {
17          for (String arg : args) {
18              ZipFile zipFile = new ZipFile(arg);
19              zipFile.stream().forEach(zipEntry -> LcmipJarAnalyser.readClass(zipFile, zipEntry));
20          }
21      }
22  
23      private static void readClass(ZipFile zipFile, ZipEntry zipEntry) {
24          try {
25              if (zipEntry.getName().endsWith(".class")) {
26                  byte[] classfile = new byte[(int) zipEntry.getSize()];
27                  DataInputStream dataInputStream = new DataInputStream(zipFile.getInputStream(zipEntry));
28                  dataInputStream.readFully(classfile);
29                  logMethods(zipEntry.getName().substring(0, zipEntry.getName().length() - 6), classfile);
30              }
31          } catch (IOException e) {
32              e.printStackTrace();
33          }
34      }
35  
36      public static void logMethods(String className, byte[] classfile) {
37          ClassReader classReader = new ClassReader(classfile);
38          ClassVisitor classVisitor = new LcmipClassVisitor(ASM5, className);
39          classReader.accept(classVisitor, ClassReader.EXPAND_FRAMES);
40      }
41  
42      private static class LcmipClassVisitor extends ClassVisitor {
43  
44          private final String className;
45  
46          public LcmipClassVisitor(int i, String className) {
47              super(i);
48              this.className = className.replaceAll("/", ".");
49          }
50  
51          public MethodVisitor visitMethod(int access, String name, String desc, String signature, String[] exceptions) {
52              System.out.println(className + "#" + name);
53              return super.visitMethod(access, name, desc, signature, exceptions);
54          }
55  
56      }
